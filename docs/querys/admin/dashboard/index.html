3. 사업자 회원​

SELECT 
    COUNT(*) as 'TOTAL',
    SUM(CASE
        WHEN si.joinstate = 0 THEN 1
        ELSE 0
    END) as '신규신청',
    SUM(CASE
        WHEN m.type = 1 THEN 1
        ELSE 0
    END) as '판매사',
    SUM(CASE
        WHEN m.type = 3 THEN 1
        ELSE 0
    END) as '광고주'
FROM
    member m inner join seller_info si on m.seq = si.member_seq
WHERE
    type <> 0 and secession = 0;

        login_log

6. 콘텐츠 순위​

SELECT 
    member.type,
    member.nick,
    content.title,
    IFNULL(viewdata.cnt, 0) AS view_count,
    IFNULL(recomenddata.cnt, 0) AS recomend_count,
    IFNULL(reportdata.cnt, 0) AS report_count,
    content.videourl
FROM
    content
        INNER JOIN
    member ON content.member_seq = member.seq
        LEFT JOIN
    (SELECT 
        content_seq,
            DATE_FORMAT(viewdatetime, '%Y-%m-%d') AS viewdate,
            COUNT(*) as cnt
    FROM
        content_view
    WHERE
        viewdatetime >= $start
            AND viewdatetime < $end
    GROUP BY content_seq , viewdate) AS viewdata ON content.seq = viewdata.cotent_seq
        LEFT JOIN
    (SELECT 
        content_seq,
            DATE_FORMAT(recommenddatetime, '%Y-%m-%d') AS recommenddate,
            COUNT(*) as cnt
    FROM
        content_recommend
    WHERE
        recommenddatetime >= $start
            AND recommenddatetime < $end
    GROUP BY content_seq , recommenddate) AS recomenddata ON content.seq = recomenddata.cotent_seq
        LEFT JOIN
    (SELECT 
        content_seq,
            DATE_FORMAT(reportdatetime, '%Y-%m-%d') AS reportdate,
            COUNT(*) as cnt
    FROM
        content_report
    WHERE
        reportdatetime >= $start
            AND reportdatetime < $end
    GROUP BY content_seq , reportdate) AS reportdata ON content.seq = reportdata.cotent_seq
order by view_count+recomend_count*100;
