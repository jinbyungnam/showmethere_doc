1. 일반회원​

SELECT 
    SUM(CASE
        WHEN secession = 0 THEN 1
        ELSE 0
    END) as 'total',
    COUNT(*) as 'join_cnt',
    SUM(CASE
        WHEN secession = 0 THEN 0
        ELSE 1
    END) as 'secession_cnt',
    SUM(CASE
        WHEN dormant = 1 THEN 1
        ELSE 0
    END) as 'dormant_cnt'
FROM
    member
WHERE
    type = 0;
항목타입코멘트
totalint전체
join_cntint가입자
secession_cntint탈퇴
dormant_cntint휴면

Table​

2. 일반회원 가입경로​

SELECT 
    COUNT(*) as 'TOTAL',
    SUM(CASE
        WHEN joindevice = 'ios' THEN 1
        ELSE 0
    END) as 'IOS',
    SUM(CASE
        WHEN joindevice = 'aos' THEN 1
        ELSE 0
    END) as 'AOS',
    SUM(CASE
        WHEN joindevice = 'web' THEN 1
        ELSE 0
    END) as 'WEB'
FROM
    member
WHERE
    type = 0 and secession = 0;
항목타입코멘트
totalint전체

3. 사업자 회원​

SELECT 
    COUNT(*) as 'TOTAL',
    SUM(CASE
        WHEN m.type = 1 and si.joinstate = 0 THEN 1
        ELSE 0
    END) as 'new_cnt',
    SUM(CASE
        WHEN m.type = 1 THEN 1
        ELSE 0
    END) as 'seller_cnt',
    SUM(CASE
        WHEN m.type = 3 THEN 1
        ELSE 0
    END) as 'advertiser_cnt'
FROM
    member m inner join seller_info si on m.seq = si.member_seq
WHERE
    type <> 0 and secession = 0;
항목타입코멘트
TOTALint전체
new_cntint신규 판매사
seller_cntint판매사 가입자
advertiser_cntint광고사 가입자

Statement​

Table​

member seller_info

4. 일별 가입자​

SELECT 
    DATE_FORMAT(joindatetime, '%Y-%m-%d') AS joindate, COUNT(*) as 'cnt'
FROM
    member
WHERE
    joindatetime >= $start
        AND joindatetime < $end
GROUP BY joindate
ORDER BY joindate;
항목타입코멘트
joindatevarchar가입일
cntint카운트

5. 일별 접속자​

SELECT 
    logindate, COUNT(seq) as 'cnt'
FROM
    (SELECT 
        DATE_FORMAT(logindatetime, '%Y-%m-%d') AS logindate, seq
    FROM
        login_log
    WHERE
        logindatetime >= $start
            AND logindatetime < $end
    GROUP BY logindate , seq)
GROUP BY logindate
ORDER BY logindate;
항목타입코멘트
logindatevarchar접속일
cntint카운트

6. 콘텐츠 순위​

SELECT 
    member.type,
    member.nick,
    content.title,
    IFNULL(viewdata.cnt, 0) AS view_count,
    IFNULL(recomenddata.cnt, 0) AS recomend_count,
    IFNULL(reportdata.cnt, 0) AS report_count,
    content.videourl
FROM
    content
        INNER JOIN
    member ON content.member_seq = member.seq
        LEFT JOIN
    (SELECT 
        content_seq,
            COUNT(*) as cnt
    FROM
        content_view
    WHERE
        viewdatetime >= $start
            AND viewdatetime < $end
    GROUP BY content_seq ) AS viewdata ON content.seq = viewdata.cotent_seq
        LEFT JOIN
    (SELECT 
        content_seq,
            COUNT(*) as cnt
    FROM
        content_recommend
    WHERE
        recommenddatetime >= $start
            AND recommenddatetime < $end
    GROUP BY content_seq ) AS recomenddata ON content.seq = recomenddata.cotent_seq
        LEFT JOIN
    (SELECT 
        content_seq,
            COUNT(*) as cnt
    FROM
        content_report
    WHERE
        reportdatetime >= $start
            AND reportdatetime < $end
    GROUP BY content_seq) AS reportdata ON content.seq = reportdata.cotent_seq
order by view_count+recomend_count*100;
항목타입코멘트
typeint회원 타입
nickvarchar닉네임
titlevarchar제목
view_countint조회수
recomend_countint추천수
report_countint신고수
videourlvarchar영상주소

Statement​

  • start:조회 시작일
  • end:조회 종료일

Table​

member content content_view content_recommend content_report
